# Dijkstra 알고리즘

> 하나의 시작 정점에서 끝 정점까지의 최단경로를 구할때 사용하는 알고리즘

경로를 통해 지나온 노드간 가중치들의 합이 가장 작은 값을 계속 갱신하며, 최단거리 가중치 합을 구하는 것이다.

- 시간복잡도 : O[N^2]
- 음이 아닌 가중치일 경우만 가능

- 시작정점이 고정되어있다.



### 인접행렬

```java
import java.util.Arrays;

public class 다익스트라_인접행렬 {

    public static void main(String[] args) {
        final int M = Integer.MAX_VALUE;

        int[][] G = {
                {0, 3, 5, M, M, M},
                {M, 0, 2, 6, M, M},
                {M, 1, 0, 4, 6, M},
                {M, M, M, 0, 2, 3},
                {3, M, M, M, 0, 6},
                {M, M, M, M, M, 0},
        };

        int N = G.length;

        int s = 0; // 시작 정점 지정

        // 그냥 넣으면 간접참조지만, 이 배열에서는 다시 사용할 일이 없으니 그냥 복사한다.
        int[] D = G[s]; // 가중치 배열, 인접행렬의 진출 차수를 기준으로 초기화
        boolean[] used = new boolean[N]; // 사용한 정점들을 체크할 배열
        used[s] = true;

        for (int i = 1; i < N; i++) { // 정점의 개수만큼 반복, 1 사이클에 정점 1개 선택됨.
            // 사용하지 않은 정점중에서, D 가중치 배열에서 최소 가중치값 정점 찾기.
            int minIdx = -1;
            int min = M;

            for (int j = 0; j < N; j++) {
                if (!used[j] && min > D[j]) {
                    min = D[j];
                    minIdx = j;
                }
            }

            used[minIdx] = true;

            // 선택한 정점을 통해서 갈 수 있는 정점의 가중치를 갱신하기.
            for (int j = 0; j < N; j++) {

                // 사용하지 않고, 사용할 가중치가 다음 j와 연결되어있다면
                if (!used[j] && G[minIdx][j] != M && D[j] > D[minIdx] + G[minIdx][j]) {
                    D[j] = D[minIdx] + G[minIdx][j];
                }
            }
        }

        System.out.println(Arrays.toString(D));

    } // end of main
} // end of class

```



### 인접리스트

```java
import java.util.ArrayList;
import java.util.Arrays;
import java.util.LinkedList;

public class 다익스트라_인접리스트 {
    public static void main(String[] args) {
        final int M = Integer.MAX_VALUE;

        ArrayList<Info>[] list = new ArrayList[6];
        for (int i = 0; i < list.length; i++) {
            list[i] = new ArrayList<>();
        }

        list[0].add(new Info(1, 3));
        list[0].add(new Info(2, 5));

        list[1].add(new Info(2, 2));
        list[1].add(new Info(3, 6));

        list[2].add(new Info(1, 1));
        list[2].add(new Info(3, 4));
        list[2].add(new Info(4, 6));

        list[3].add(new Info(4, 2));
        list[3].add(new Info(5, 3));

        list[4].add(new Info(0, 3));
        list[4].add(new Info(5, 6));

        LinkedList<Info> q = new LinkedList<>();
        int start = 0;

        int[] d = new int[6];
        for (int i = 0; i < d.length; i++) {
            if (i == start) continue;
            d[i] = M;
        }

        q.add(new Info(start, 0));

        while(!q.isEmpty()) {
            Info now = q.poll();

            for (int i = 0; i < list[now.idx].size(); i++) {
                int next = list[now.idx].get(i).idx;
                int nextDis = now.dis + list[now.idx].get(i).dis;

                if(d[next] > nextDis) {
                    d[next] = nextDis;
                    q.add(new Info(next, nextDis));
                }
            }
        }

        System.out.println(Arrays.toString(d));
    }

    public static class Info{
        int idx, dis;

        public Info(int idx, int dis) {
            this.idx = idx;
            this.dis = dis;
        }
    }
}
```
